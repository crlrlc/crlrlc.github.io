源码:
1)/给三个数简单排序/
#include<stdio.h>
int main(void)
{
int a, b, c, d;
printf("请输入三个数让系统对其从大到小排序：");
scanf_s("%d %d %d", &a, &b, &c);

if (a < b)
{
	d = a;
	a = b;
	b = d;
}

if (a < c)
{
	d = a;
	a = c;
	c = d;
}

if (b < c)    //千万注意if后面不能有；！！！！！
{
	d = b;
	b = c;
	c = d;
}

printf("%d %d %d\n", a, b, c);
return 0;
}
