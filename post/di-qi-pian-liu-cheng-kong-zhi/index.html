源码:
1)/*
目的：流程控制：程序执行的顺序，顺序执行，选择，循环
学习if的用法
*/
#include<stdio.h>
int main(void)
{
if (3)
printf("aaaa\n");//输出
if (0)
printf("bbbb\n");//不输出
if (0 == 0)
printf("ccccc\n");//输出

if (1 > 2)
	printf("ashdkajhsdk\n");
	printf("bbbb\n");//结果只输出bbbb

if (3 > 2)
{
	printf("ashdkajhsdk\n");
	printf("bbbb\n");//两个同时输出或不输出
}




return 0;

}
/*
if！！！在不加括号！！！！的情况下只能控制一个语句的执行
