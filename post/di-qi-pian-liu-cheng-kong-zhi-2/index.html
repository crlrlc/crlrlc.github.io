1)#include<stdio.h>
int main(void)
{
if (1 > 2)
printf("aaaa\n");//会输出
else
printf("bbbb\n");//不输出
printf("cccc\n");//会输出，有2个语句

int asd;
asd = -1231;
if (asd > 0)
	printf("有两个解\n");
//printf("哈哈哈哈");注意这行不能有，因为else if要与if相匹配
else if (asd == 0)
	printf("只有一个解\n");//注意，如果if成立，则else if不会执行。
else
	printf("无解\n");




return 0;
}

if的格式及优先级：
if(1)
a;
else if(2)

else if(3)
c;		【如果这个成立，则1，2都不成立】
else(这里不可以有表达式！！！！)
d;		【如果这个成立则1，2，3都不成立】
【1，2，3都成了的话只执行a】

    xxxx;
